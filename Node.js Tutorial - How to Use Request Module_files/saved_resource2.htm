                                      Chad Kruse Andrew Minton 2 months ago

                                      Your use case sounds similar to one I had a few days ago. You likely need to do two requests. One to the REST service to capture the PDF url, and the next to actually download the file.

                                      Here's some sample code in case it helps:

                                      var request = require('request'),
    fs = require('graceful-fs'),
    url = require('url'),
    es = require('event-stream'),
    JSONStream = require('JSONStream');

var index = 'https://s3.amazonaws.com/irs-form-990/index.json';

request({url: index})
  .pipe(JSONStream.parse(['AllFilings',true]))
  .pipe(es.mapSync(function (data) {
    if(data.URL && data.URL.length > 0 && data.FormType == "990PF") {
      var filename = url.parse(data.URL).pathname;
      request(data.URL)
        .on('error', function (e) {
          console.log(e);
        })
        .pipe(fs.createWriteStream('./private' + filename))
    }
    return data;
  })
  .on('error', function(e) {
      console.error(e);
  })
);
